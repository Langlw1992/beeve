Progress 组件用于展示当前任务的进度。

## 基础用法

<ProgressBasic client:only="solid-js" />

```tsx
import { Progress } from '@beeve/ui'

<Progress percent={60} />
```

## 进度条变体

<ProgressVariants client:only="solid-js" />

```tsx
<Progress percent={60} />                    {/* 默认 */}
<Progress percent={100} status="success" />  {/* 成功 */}
<Progress percent={75} status="active" />    {/* 活动 */}
<Progress percent={30} status="exception" /> {/* 错误 */}
```

## 进度条尺寸

<ProgressSizes client:only="solid-js" />

```tsx
<Progress percent={60} size="sm" />
<Progress percent={60} size="md" />
<Progress percent={60} size="lg" />
```

## 动画进度条

<ProgressAnimated client:only="solid-js" />

```tsx
import { createSignal, onCleanup } from 'solid-js'

const [percent, setPercent] = createSignal(0)

const interval = setInterval(() => {
  setPercent((v) => (v >= 100 ? 0 : v + 10))
}, 500)

onCleanup(() => clearInterval(interval))

<Progress percent={percent()} />
```

## 不确定进度

<ProgressIndeterminate client:only="solid-js" />

```tsx
<Progress indeterminate />
```

## 带标签进度条

<ProgressWithLabel client:only="solid-js" />

```tsx
<div class="space-y-2">
  <div class="flex justify-between text-sm">
    <span>上传进度</span>
    <span>45%</span>

  <Progress percent={45} />
</div>
```

## API

### ProgressProps

| 属性 | 类型 | 默认值 | 说明 |
|------|------|--------|------|
| `percent` | `number` | `0` | 当前进度百分比 (0-100) |
| `type` | `'line' \| 'circle'` | `'line'` | 进度条类型 |
| `status` | `'normal' \| 'success' \| 'exception' \| 'active'` | `'normal'` | 进度条状态 |
| `size` | `'sm' \| 'md' \| 'lg'` | `'md'` | 进度条尺寸 |
| `showInfo` | `boolean` | `true` | 是否显示进度信息 |
| `format` | `(percent: number) => JSX.Element \| string` | - | 自定义进度信息格式 |
| `strokeWidth` | `number` | - | 进度条宽度（仅 circle 类型） |
| `strokeColor` | `string` | - | 进度条颜色 |
| `trailColor` | `string` | - | 轨道颜色 |
| `class` | `string` | - | 自定义类名 |
